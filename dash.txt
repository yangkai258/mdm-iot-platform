<template>
  <div class="page-container">
    <!-- 闈㈠寘灞� -->
    <a-breadcrumb class="breadcrumb">
      <a-breadcrumb-item>棣栭〉</a-breadcrumb-item>
      <a-breadcrumb-item>宸ヤ綔鍙�</a-breadcrumb-item>
    </a-breadcrumb>

    <!-- 缁熻�″崱鐗� -->
    <a-row :gutter="16" class="stats-row">
      <a-col :span="6">
        <a-card class="stat-card">
          <a-statistic title="鎬昏�惧�囨暟" :value="stats.total_devices">
            <template #prefix>馃摫</template>
          </a-statistic>
        </a-card>
      </a-col>
      <a-col :span="6">
        <a-card class="stat-card">
          <a-statistic title="鍦ㄧ嚎璁惧��" :value="stats.online_devices" :value-style="{ color: '#52c41a' }">
            <template #prefix>馃煝</template>
          </a-statistic>
        </a-card>
      </a-col>
      <a-col :span="6">
        <a-card class="stat-card">
          <a-statistic title="绂荤嚎璁惧��" :value="stats.offline_devices" :value-style="{ color: '#ff4d4f' }">
            <template #prefix>馃敶</template>
          </a-statistic>
        </a-card>
      </a-col>
      <a-col :span="6">
        <a-card class="stat-card">
          <a-statistic title="寰呭�勭悊鍛婅��" :value="stats.pending_alerts" :value-style="{ color: '#faad14' }">
            <template #prefix>鈿狅笍</template>
          </a-statistic>
        </a-card>
      </a-col>
    </a-row>

    <!-- 鏁版嵁灞曠ず鍖� -->
    <a-row :gutter="16" style="margin-top: 16px;">
      <a-col :span="12">
        <a-card title="璁惧�囩姸鎬佸垎甯�">
          <div class="chart-placeholder">
            <a-progress type="circle" :percent="onlineRate" :width="150">
              <template #formatter>
                <span style="font-size: 24px;">{{ onlineRate }}%</span>
              </template>
            </a-progress>
            <p>璁惧�囧湪绾跨巼</p>
          </div>
        </a-card>
      </a-col>
      <a-col :span="12">
        <a-card title="鏈�杩戝憡璀�">
          <a-list :data-source="recentAlerts">
            <template #empty>
              <a-empty description="鏆傛棤鍛婅��" />
            </template>
            <a-list-item v-for="alert in recentAlerts" :key="alert.id">
              <a-list-item-meta>
                <template #avatar>
                  <a-avatar :style="{ backgroundColor: getSeverityColor(alert.severity) }">
                    {{ alert.severity }}
                  </a-avatar>
                </template>
                <template #title>{{ alert.message }}</template>
                <template #description>{{ alert.device_id }} - {{ alert.created_at }}</template>
              </a-list-item-meta>
            </a-list-item>
          </a-list>
        </a-card>
      </a-col>
    </a-row>
  </div>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'

const stats = ref({
  total_devices: 0,
  online_devices: 0,
  offline_devices: 0,
  total_alerts: 0,
  pending_alerts: 0
})

const recentAlerts = ref([])

const onlineRate = computed(() => {
  if (stats.value.total_devices === 0) return 0
  return Math.round((stats.value.online_devices / stats.value.total_devices) * 100)
})

const getSeverityColor = (severity) => {
  const colors = { 1: '#52c41a', 2: '#1890ff', 3: '#faad14', 4: '#ff4d4f' }
  return colors[severity] || '#999'
}

const loadStats = async () => {
  try {
    const token = localStorage.getItem('token')
    const res = await fetch('/api/v1/dashboard/stats', {
      headers: { 'Authorization': `Bearer ${token}` }
    })
    const data = await res.json()
    if (data.code === 0) {
      const d = data.data || {}
      // 鍏煎�瑰悗绔�鍙�鑳借繑鍥炵殑涓嶅悓瀛楁�靛悕
      stats.value = {
        total_devices: d.total_devices ?? d.total ?? d.device_total ?? 0,
        online_devices: d.online_devices ?? d.online ?? d.device_online ?? 0,
        offline_devices: d.offline_devices ?? d.offline ?? d.device_offline ?? 0,
        total_alerts: d.total_alerts ?? d.alerts_total ?? d.alert_total ?? 0,
        pending_alerts: d.pending_alerts ?? d.pending ?? d.alerts_pending ?? 0
      }
    }

    // Load recent alerts
    const alertRes = await fetch('/api/v1/alerts?status=1', {
      headers: { 'Authorization': `Bearer ${token}` }
    })
    const alertData = await alertRes.json()
    if (alertData.code === 0) {
      recentAlerts.value = (alertData.data?.list || alertData.data || []).slice(0, 5)
    }
  } catch (e) {
    console.error('鍔犺浇缁熻�″け璐�:', e)
  }
}

onMounted(() => {
  loadStats()
})
</script>
